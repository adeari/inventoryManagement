create table roles (id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, name varchar(500),description varchar(1000));
create table user (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
username varchar(500),
email varchar(500),
first_name varchar(500),
last_name varchar(500),
password varchar(500),
company_name varchar(500),
phone varchar(500),
address varchar(500),
roles_id integer,
created_at timestamp,
modified_at timestamp
);
create table items(
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
name varchar(500),
description varchar(500),
quantity integer,
sku varchar(500),
created_at timestamp,
modified_at timestamp,
category_id integer
);
create table inventories (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
item_id integer,
stock_in integer,
stock_out integer,
created_at timestamp,
modified_at timestamp,
transaction_id integer
);
create table inventory_metas ( 
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
inventory_id integer,
keyData varchar(500),
valueData longvarchar
);
create table inventory_prices(
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
inventory_id integer,
price_in integer,
price_out integer
);
create table projects (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
client_id integer ,
title varchar(500) ,
description varchar(1000) ,
location varchar(1000) ,
leader_id integer ,
status integer ,
starts_on timestamp ,
completed_on timestamp ,
due_date timestamp ,
created_at timestamp ,
modified_at timestamp ,
created_by integer 
);
create table project_metas (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
project_id integer,
keyData varchar(500),
valueData longvarchar
);
create table assignments (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
user_id integer,
project_id integer
);
create table project_objects (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
project_id integer,
type varchar(500),
inventory_id integer,
inventory_qty integer,
created_at timestamp,
modified_at timestamp,
created_by integer
);
create table accounts (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
unique_id varchar(20),
name varchar(50)
);
create table cache_products_in_project(
id integer,
user_id integer,
name varchar(500),
qty integer
);
create table cache_people_in_project (
id_people integer,
id_user integer,
people_name varchar(1001)
);
create table item_project (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
project_id integer,
type varchar(500),
item_id integer,
inventory_qty integer,
created_at timestamp,
modified_at timestamp,
created_by integer
);
create table transactions (
id integer  GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
title varchar(500),
description varchar(1000),
value integer,
created_at timestamp,
modified_at timestamp,
created_by integer,
account_id integer,
project_id integer
);
